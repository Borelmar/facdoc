pkglibs = pkg-config --cflags --libs gtk+-3.0 webkit2gtk-4.0
CFLAGS = -Wall -D_DEBUG_
LDFLAGS =

all: setup

setup: src/main.c
	mkdir -p ./build/icemoon
	mkdir ./build/icemoon/css
	mkdir ./build/icemoon/img
	cp ./src/css/gtk.css ./build/icemoon/css/gtk.css
	cp ./src/img/icon-256.png ./build/icemoon/img/icon-256.png
	gcc $(LDFLAGS) src/main.c `$(pkglibs)` -o ./build/icemoon/icemoon $(CFLAGS)

clean:
	rm -rf ./build
